confidence = function(technique) {
	data = read.table("data.txt", header=TRUE, sep=",")
	sub=subset(data,Technique==technique)$Time
	ecartType = sd(sub)
	n = sqrt(length(sub))
	ci = 1.96*(ecartType/n)
	return(ci)
}
